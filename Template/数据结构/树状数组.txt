template <typename T>
struct Fenwick {
    int n;
    std::vector<T> a;
    
    Fenwick(int n_ = 0) {
        init(n_);
    }
    
    void init(int n_) {
        n = n_;
        a.assign(n + 1, T{});//初始化大小为n + 1
    }

    void add(int x, const T &v) {//单点修改
        while(x <= n)
            a[x] += v, x += x & -x;
    }
    
    T sum(int x) {//查询1~x的和
        T ans{};
        while(x > 0)
            ans += a[x], x -= x & -x;
        return ans;
    }
    //1-based
    T query(int l, int r) {
        return sum(r) - sum(l - 1);
    }
};
